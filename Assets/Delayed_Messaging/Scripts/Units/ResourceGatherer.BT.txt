tree "Root"
	fallback
		tree "Gather"
		tree "Idle"

tree "Gather"
    while Gather
        repeat
            sequence
                tree "Collect"
                tree "Deposit"
    
tree "Collect"
    sequence
        MoveToResource
        GatherResource

tree "Deposit"
    sequence
        MoveToDeposit
        DepositResource
    			
tree "Idle"
	while
		sequence
			not Gather
		repeat Succeed
